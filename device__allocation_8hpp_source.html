57template<class T, reuse_pages ReusePagesFlag, page_size_t PageSize>
58class device_allocation<
59 pagination_type::contiguous,
60 T,
61 ReusePagesFlag,
62 PageSize> {
63 public:
64 device_allocation(
65 device_id_t device_id,
66 const std::vector<page_index_t>& indices,
67 const std::vector<
68 page_handle<page_handle_type::weak, PageSize>>& /*unused*/
69 ) {
70 auto page_count = static_cast<page_count_t>(indices.size());
71 auto bytes_per_page = page_traits<T>::allocated_bytes_per_page;
72 allocation_handle alloc_handle;
73 auto device = sclx::device::get_devices()[device_id];
74 sycl::queue queue{device};
75 alloc_handle.data_ = sclx::make_unique<sclx::byte_array>(
76 queue,
77 sclx::usm::alloc::device,
78 static_cast<std::size_t>(page_count * bytes_per_page)
79 );
80 auto& pages = alloc_handle.pages_;
81 for (page_index_t pidx = 0; pidx < page_count; ++pidx) {
82 auto page_index = indices[pidx];
83 auto page_ptr = alloc_handle.data_.get() + pidx * bytes_per_page;
84 auto device_page_ptr = std::make_shared<device_page<PageSize>>(
85 page_ptr,
86 device_id,
87 page_index,
88 bytes_per_page
89 );
90 pages.emplace_back(
91 make_page_handle<page_handle_type::strong, PageSize>(
92 device_page_ptr
93 )
94 );
95 }
96 anchor_ = make_access_anchor(device_id, std::move(alloc_handle));
97 }
98
99 device_allocation(
100 device_id_t device_id,
101 page_index_t first,
102 page_index_t last,
103 const std::vector<
104 page_handle<page_handle_type::weak, PageSize>>& /*unused*/
105 ) {
106 auto page_count = last - first;
107 auto bytes_per_page = page_traits<T>::allocated_bytes_per_page;
108 allocation_handle alloc_handle;
109 auto device = sclx::device::get_devices()[device_id];
110 sycl::queue queue{device};
111 alloc_handle.data_ = sclx::make_unique<sclx::byte_array>(
112 queue,
113 sclx::usm::alloc::device,
114 page_count * bytes_per_page
115 );
116 auto& pages = alloc_handle.pages_;
117 for (page_index_t pidx = 0; pidx < page_count; ++pidx) {
118 auto page_index = first + pidx;
119 auto page_ptr = alloc_handle.data_.get() + pidx * bytes_per_page;
120 auto device_page_ptr = std::make_shared<device_page<PageSize>>(
121 page_ptr,
122 device_id,
123 page_index,
124 bytes_per_page
125 );
126 pages.emplace_back(
127 make_page_handle<page_handle_type::strong, PageSize>(
128 device_page_ptr
129 )
130 );
131 }
132 anchor_ = make_access_anchor(device_id, std::move(alloc_handle));
133 }
