66 template<class F>

68 : task_{std::make_shared<std::packaged_task<function_type>>(

99 void async_execute() const override {
100 auto& args_ptr = args_ptr_;
101 auto& task = task_;
102 auto metadata = metadata_.get_view<access_mode::write>();
103 std::thread exec_thread{
104 [metadata = std::move(metadata), args_ptr, task] {
105 typed_impl::apply(*task, *args_ptr);
106 for (const auto& dependent_task :
107 metadata.access().dependent_tasks) {
108 dependent_task.impl_->decrease_dependency_count();
109 }
110 metadata.access().has_completed = true;
111 }
112 };
113 exec_thread.detach();
114 }

116 template<std::uint8_t N = 0, class Task, class Tuple, class... OArgs>
117 static void apply(Task& task, Tuple& args, OArgs&&... oargs) {
118 if constexpr (sizeof...(Args) == 0) {
119 return task();
120 }
121
122 if constexpr (N == sizeof...(Args)) {
123 return task(std::forward<OArgs>(oargs)...);
124 } else {
125 if constexpr (std::get<N>(is_rvalue_ref_args_)) {
126 return apply<N + 1>(
127 task,
128 args,
129 std::forward<OArgs>(oargs)...,
130 std::move(std::get<N>(args))
131 );
132 } else {
133 return apply<N + 1>(
134 task,
135 args,
136 std::forward<OArgs>(oargs)...,
137 std::get<N>(args)
138 );
139 }
140 }
141 }
142
143 std::shared_ptr<std::packaged_task<function_type>> task_;
144 std::shared_ptr<std::tuple<std::remove_reference_t<Args>...>> args_ptr_;
145 static constexpr auto is_rvalue_ref_args_
146 = std::make_tuple(std::is_rvalue_reference_v<Args>...);
147};

149struct task_factory {
150 template<class F, class... Args>
151 static auto create_task(F&& func, Args&&... args)
152 -> typed_task<std::invoke_result_t<F, Args...>> {
153 using task_t = typed_task<std::invoke_result_t<F, Args...>>;
154 using typed_impl_t = typename task_t::template typed_impl<
155 decltype(std::forward<Args>(args))...>;
156 return task_t{std::make_unique<typed_impl_t>(
157 std::forward<F>(func),
158 std::forward<Args>(args)...
159 )};
160 }
161};

163template<class F, class... Args>
164auto create_task(F&& func, Args&&... args)
165 -> typed_task<std::invoke_result_t<F, Args...>> {
166 return task_factory::create_task(
167 std::forward<F>(func),
168 std::forward<Args>(args)...
169 );
170}
