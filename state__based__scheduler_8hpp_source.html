72 template<class Action, class... Args>
73 struct action_wrapper {
74 Action action;
75 std::tuple<Args...> args;
76
77 explicit action_wrapper(Action&& action, Args&&... args)
78 : action{std::move<Action>(action)},
79 args{std::forward<Args>(args)...} {}
80
81 action_wrapper(const action_wrapper&) = delete;
82 auto operator=(const action_wrapper&) -> action_wrapper& = delete;
83 action_wrapper(action_wrapper&&) = default;
84 auto operator=(action_wrapper&&) -> action_wrapper& = default;
85
86 void operator()() const { std::apply(action, args); }
87
88 ~action_wrapper() = default;
89 };

91 template<class Action, class... Args>
92 struct action_definition {
93 state_label entry_requirement;
94 inputs_type inputs_on_test_entry;
95 inputs_type inputs_on_exit;
96 action_wrapper<Action, Args...> action;
97 };
98
99 public:
100 template<class Action, class... Args>
101 auto submit(
102 const state_label& entry_requirement,
103 inputs_type inputs_on_test_entry,
104 inputs_type inputs_on_exit,
105 Action&& action,
106 Args&&... args
107 ) -> sclx::event {
108 using action_wrapper_type = action_wrapper<Action, Args...>;
109
110 auto metadata_view = metadata_guard_.get_view();
111 auto queue_empty_on_submit
112 = metadata_view.access().action_queue_.empty();
113
114 auto wrapped_action = action_wrapper_type{
115 std::forward<Action>(action),
116 std::forward<Args>(args)...
117 };
118
119 std::promise<void> action_signal;
120 auto action_signal_future = action_signal.get_future();
121 auto promised_action_def = action_definition{
122 entry_requirement,
123 inputs_on_test_entry,
124 inputs_on_exit,
125 std::move(wrapped_action)
126 };
127 metadata_view.access().action_signal_queue_.emplace(
128 std::move(action_signal),
129 true
130 );
131
132 auto action_event = queue_.submit([&](sycl::handler& cgh) {
133 const auto& metadata_guard = metadata_guard_;
134 host_task(
135 cgh,
136 [action_signal_future = std::move(action_signal_future),
137 promised_action_def = std::move(promised_action_def),
138 metadata_guard] mutable {
139 // wait for the action to be signaled
140 // by the scheduler
141 action_signal_future.get();
142
143 auto action_metadata_view = metadata_guard.get_view();
144 action_metadata_view.unlock();
145 while (true) {
146 action_metadata_view.lock();
147
148 auto& metadata = action_metadata_view.access();
149
150 // the state machine is transitioned each loop
151 // by the inputs testing the entry requirement
152 //
153 // making the entry requirement a callable was
154 // considered, in which case the state machine would've
155 // only been transitioned if the requirement was
156 // satisfied which would occur once another running
157 // action transitioned the state machine to a valid
158 // state on exit. this would allow actions to have entry
159 // requirements that took a union of different states
160 //
161 // but it made more sense to keep the entry requirement
162 // as a simple state label check, instead offloading
163 // the logic to the state machine itself, keeping
164 // the spirit of a state-based scheduler intact
165 metadata.state_machine_.transition_states(
166 promised_action_def.inputs_on_test_entry
167 );
168
169 if (metadata.state_machine_.current_state()
170 == promised_action_def.entry_requirement) {
171 break;
172 }
173
174 action_metadata_view.unlock();
175
176 std::this_thread::yield();
177 }
178
179 action_metadata_view.access().action_signal_queue_.pop();
180 action_metadata_view.unlock();
181 promised_action_def.action();
182
183 action_metadata_view.lock();
184 action_metadata_view.access()
185 .state_machine_.transition_states(
186 promised_action_def.inputs_on_exit
187 );
188 }
189 );
190 });
191
192 // by calling execute action for every action submitted
193 // we can ensure that each action will get executed, even if
194 // the following call doesn't execute the action submitted
195 // by the current call
196 execute_next_action();
197
198 return action_event;
199 }
200
201 private:
202 void execute_next_action() {
203 auto metadata_view = metadata_guard_.get_view();
204 metadata_view.unlock();
205 while (true) {
206 metadata_view.lock();
207 auto& metadata = metadata_view.access();
208
209 if (auto& action_signal_queue = metadata.action_signal_queue_;
210 action_signal_queue.front().action_signal_valid) {
211 action_signal_queue.front().action_signal.set_value();
212 action_signal_queue.front().action_signal_valid = false;
213 return;
214 }
215
216 metadata_view.unlock();
217
218 std::this_thread::yield();
219 }
220 }
221
222 concurrent_guard<scheduler_metadata> metadata_guard_{
223 std::make_unique<scheduler_metadata>()
224 };
225
226 sycl::queue queue_;
227};
