41template<page_size_t PageSize = default_page_size>
42class device_page_table : public page_table_interface<PageSize> {
43 public:
44 using interface = sclx::detail::page_table_interface<PageSize>;
45 using page_handle = typename interface::page_handle;
46 using iterator = typename interface::iterator;
47
48 device_page_table(const device& device, page_count_t page_count)
49 : queue_(device),
50 device_id_(find_device(device)),
51 pages_(page_count),
52 host_staging_page_data_(page_count),
53 device_accessible_page_data_(
54 sclx::make_unique<sclx::array_type<page_data<PageSize>>>(
55 queue_,
56 sclx::usm::alloc::device,
57 page_count
58 )
59 ) {}
60
61 auto map_page(page_handle page) -> sclx::event override {
62 auto page_lock = page.lock();
63 if (page_lock.index() == sclx::invalid_page) {
64 throw std::runtime_error(
65 "Cannot map an invalid page. This error indicates a bug in "
66 "Scalix, please report it."
67 );
68 }
69 return pages_[page_lock.index()].lock().replace_with(queue_, page_lock);
70 }
71
72 auto device_id() -> std::variant<device_id_t, sclx::mpi_device> override {
73 return device_id_;
74 }
75
76 auto unmap_invalid_pages() -> sclx::event override {
77 for (auto& page : pages_) {
78 if (page.lock().is_valid()) {
79 page.release();
80 }
81 }
82
83 return {};
84 }
85
86 auto begin() -> iterator override { return pages_.begin(); }
87
88 auto end() -> iterator override { return pages_.end(); }
89
90 auto make_table_host_accessible() -> sclx::event override {
91 auto copy_queue = queue_.memcpy(
92 host_staging_page_data_.data(),
93 device_accessible_page_data_.get(),
94 pages_.size() * sizeof(page_data<PageSize>)
95 );
96 return queue_.submit([&](sycl::handler& cgh) {
97 cgh.depends_on(copy_queue);
98 cgh.host_task([&]() {
99 for (std::size_t i = 0; i < pages_.size(); ++i) {
100 pages_[i].lock().set_write_bit(
101 host_staging_page_data_[i].write_bit
102 );
103 }
104 });
105 });
106 }
107
108 auto make_table_device_accessible() -> sclx::event override {
109 auto staging_event = queue_.submit([&](sycl::handler& cgh) {
110 cgh.host_task([=, this]() {
111 for (std::size_t i = 0; i < pages_.size(); ++i) {
112 auto page_lock = pages_[i].lock();
113 if (!page_lock.is_mpi_local()) {
114 throw std::runtime_error(
115 "Device page handles should not exist on other MPI "
116 "nodes. "
117 "Assuming "
118 "bad page. This error indicates a bug in Scalix, "
119 "please "
120 "report it."
121 );
122 }
123
124 auto data_variant = page_lock.data();
125 auto data = std::get<sclx::byte*>(data_variant);
126 host_staging_page_data_[i].data = data;
127 auto write_bit_variant = page_lock.write_bit();
128 auto write_bit
129 = std::get<sclx::write_bit_t>(write_bit_variant);
130 host_staging_page_data_[i].write_bit = write_bit;
131 }
132 });
133 });
134
135 return queue_.submit([&](sycl::handler& cgh) {
136 cgh.depends_on(staging_event);
137 cgh.memcpy(
138 device_accessible_page_data_.get(),
139 host_staging_page_data_.data(),
140 pages_.size() * sizeof(page_data<PageSize>)
141 );
142 });
143 }
144
145 private:
146 sycl::queue queue_;
147 device_id_t device_id_;
148 std::vector<page_handle> pages_;
149 std::vector<page_data<PageSize>> host_staging_page_data_;
150 sclx::unique_ptr<sclx::array_type<page_data<PageSize>>>
151 device_accessible_page_data_;
152};
