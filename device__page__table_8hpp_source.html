41template<page_size_t PageSize = default_page_size>
42class device_page_table : public page_table_interface<PageSize> {
43 public:
44 using interface = sclx::detail::page_table_interface<PageSize>;
45 using weak_page_handle = typename interface::weak_page_handle;
46 using strong_page_handle = typename interface::strong_page_handle;
47 using iterator = typename interface::iterator;
48
49 device_page_table(const device& device, page_count_t page_count)
50 : queue_(device),
51 device_id_(find_device(device)),
52 pages_(page_count),
53 host_staging_page_data_(page_count),
54 device_accessible_page_data_(
55 sclx::make_unique<sclx::array_type<page_data<PageSize>>>(
56 queue_,
57 sclx::usm::alloc::device,
58 page_count
59 )
60 ) {}
61
62 auto map_page(strong_page_handle page) -> sclx::event override {
63 if (page.index() == sclx::invalid_page) {
64 throw std::runtime_error(
65 "Cannot map an invalid page. This error indicates a bug in "
66 "Scalix, please report it."
67 );
68 }
69 return pages_[page.index()].lock().replace_with(queue_, page);
70 }
71
72 auto device_id() -> std::variant<device_id_t, sclx::mpi_device> override {
73 return device_id_;
74 }
75
76 auto unmap_invalid_pages() -> sclx::event override {
77 for (auto& page : pages_) {
78 if (!page.lock().is_valid()) {
79 page.release();
80 }
81 }
82
83 return {};
84 }
85
86 auto begin() -> iterator override { return pages_.begin(); }
87
88 auto end() -> iterator override { return pages_.end(); }
89
90 auto pages() const -> const std::vector<weak_page_handle>& override {
91 return pages_;
92 }
93
94 auto make_table_host_accessible() -> sclx::event override {
95 auto copy_queue = queue_.memcpy(
96 host_staging_page_data_.data(),
97 device_accessible_page_data_.get(),
98 pages_.size() * sizeof(page_data<PageSize>)
99 );
100 return queue_.submit([&](sycl::handler& cgh) {
101 cgh.depends_on(copy_queue);
102 cgh.host_task([&]() {
103 for (std::size_t i = 0; i < pages_.size(); ++i) {
104 pages_[i].lock().set_write_bit(
105 host_staging_page_data_[i].write_bit
106 );
107 }
108 });
109 });
110 }
111
112 auto make_table_device_accessible() -> sclx::event override {
113 auto staging_event = queue_.submit([&](sycl::handler& cgh) {
114 cgh.host_task([=, this]() {
115 for (std::size_t i = 0; i < pages_.size(); ++i) {
116 auto page_lock = pages_[i].lock();
117 if (!page_lock.is_mpi_local()) {
118 throw std::runtime_error(
119 "Device page handles should not exist on other MPI "
120 "nodes. "
121 "Assuming "
122 "bad page. This error indicates a bug in Scalix, "
123 "please "
124 "report it."
125 );
126 }
127
128 auto data_variant = page_lock.data();
129 auto data = std::get<sclx::byte*>(data_variant);
130 host_staging_page_data_[i].data = data;
131 auto write_bit_variant = page_lock.write_bit();
132 auto write_bit
133 = std::get<sclx::write_bit_t>(write_bit_variant);
134 host_staging_page_data_[i].write_bit = write_bit;
135 }
136 });
137 });
138
139 return queue_.submit([&](sycl::handler& cgh) {
140 cgh.depends_on(staging_event);
141 cgh.memcpy(
142 device_accessible_page_data_.get(),
143 host_staging_page_data_.data(),
144 pages_.size() * sizeof(page_data<PageSize>)
145 );
146 });
147 }
148
149 private:
150 sycl::queue queue_;
151 device_id_t device_id_;
152 std::vector<weak_page_handle> pages_;
153 std::vector<page_data<PageSize>> host_staging_page_data_;
154 sclx::unique_ptr<sclx::array_type<page_data<PageSize>>>
155 device_accessible_page_data_;
156};
