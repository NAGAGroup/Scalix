40template<page_size_t PageSize = default_page_size>
41class host_page_table : public page_table_interface<PageSize> {
42 public:
43 using interface = sclx::detail::page_table_interface<PageSize>;
44 using weak_page_handle = typename interface::weak_page_handle;
45 using strong_page_handle = typename interface::strong_page_handle;
46 using iterator = typename interface::iterator;
47
48 explicit host_page_table(page_count_t page_count)
49 : pages_(page_count),
50 simplified_page_data_(page_count) {}
51
52 auto map_page(strong_page_handle page) -> sclx::event override {
53 if (page.index() == sclx::invalid_page) {
54 throw std::runtime_error(
55 "Cannot map an invalid page. This error indicates a bug in "
56 "Scalix, please report it."
57 );
58 }
59 return pages_[page.index()].lock().replace_with(queue_, page);
60 }
61
62 auto device_id() -> std::variant<device_id_t, sclx::mpi_device> override {
63 return sclx::host_device_id;
64 }
65
66 auto unmap_invalid_pages() -> sclx::event override {
67 for (auto& page : pages_) {
68 if (!page.lock().is_valid()) {
69 page.release();
70 }
71 }
72
73 return {};
74 }
75
76 auto begin() -> iterator override { return pages_.begin(); }
77
78 auto end() -> iterator override { return pages_.end(); }
79
80 auto pages() const -> const std::vector<weak_page_handle>& override {
81 return pages_;
82 }
83
84 auto make_table_host_accessible() -> sclx::event override {
85 return queue_.submit([&](sycl::handler& cgh) {
86 cgh.host_task([&]() {
87 for (std::size_t i = 0; i < pages_.size(); ++i) {
88 pages_[i].lock().set_write_bit(
89 simplified_page_data_[i].write_bit
90 );
91 }
92 });
93 });
94 }
95
96 auto make_table_device_accessible() -> sclx::event override {
97 return queue_.submit([&](sycl::handler& cgh) {
98 cgh.host_task([&]() {
99 for (std::size_t i = 0; i < pages_.size(); ++i) {
100 auto page_lock = pages_[i].lock();
101 if (!page_lock.is_mpi_local()) {
102 throw std::runtime_error(
103 "Device page handles should not exist on other MPI "
104 "nodes. "
105 "Assuming "
106 "bad page. This error indicates a bug in Scalix, "
107 "please "
108 "report it."
109 );
110 }
111 simplified_page_data_[i].write_bit
112 = std::get<sclx::write_bit_t>(page_lock.write_bit());
113 simplified_page_data_[i].data
114 = std::get<sclx::byte*>(page_lock.data());
115 }
116 });
117 });
118 }
119
120 private:
121 sycl::queue queue_{};
122 std::vector<weak_page_handle> pages_{};
123 std::vector<page_data<PageSize>> simplified_page_data_{};
124};
