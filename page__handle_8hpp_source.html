199 auto
200 copy_from(sycl::queue queue, page_handle src, page_copy_rules rules = {})
201 -> sclx::event {
202 auto skip_copy_variant = should_skip_copy(*this, src, rules);
203 if (src.is_mpi_local()) {
204 if (std::get<bool>(skip_copy_variant)) {
205 return {};
206 }
207 } else {
208 auto skip_copy_future
209 = std::get<std::future<bool>>(skip_copy_variant).share();
210 auto skip_copy_ptr = std::make_shared<bool>();
211 auto skip_copy_event = queue.submit([&](sycl::handler& cgh) {
212 cgh.host_task([=]() { *skip_copy_ptr = skip_copy_future.get(); }
213 );
214 });
215 return queue.submit([&](sycl::handler& cgh) {
216 cgh.depends_on(skip_copy_event);
217 cgh.host_task([src, skip_copy_ptr, queue, this]() {
218 if (*skip_copy_ptr) {
219 return;
220 }
221 this->impl_->copy_from(queue, src.impl_).wait_and_throw();
222 });
223 });
224 }
225
226 return impl_->copy_from(queue, src.impl_);
227 }
228
229 [[nodiscard]] auto is_mpi_local() const -> bool {
230 if (!is_valid()) {
231 return true;
232 }
233 return impl_->is_mpi_local();
234 }

236 auto replace_with(sycl::queue queue, page_handle new_page) -> sclx::event {
237 if (!this->is_valid()) {
238 *this = std::move(new_page);
239 return {};
240 }
241 if (!new_page.is_valid()) {
242 throw std::runtime_error(
243 "Cannot replace a page with an invalid page"
244 );
245 }
246 auto& old_page = *this;
247 auto copy_event = new_page.copy_from(
248 queue,
249 old_page,
250 page_copy_rules{
251 .expect_valid_src = false,
252 .expect_valid_dst = false
253 }
254 );
255 if (old_page.is_mpi_local()) {
256 auto write_bit_variant = old_page.write_bit();
257 auto write_bit = std::get<write_bit_t>(write_bit_variant);
258 new_page.set_write_bit(write_bit);
259 *this = std::move(new_page);
260 return copy_event;
261 }
262
263 auto shared_write_bit = std::make_shared<write_bit_t>();
264 auto write_bit_retrieval_event = queue.submit([&](sycl::handler& cgh) {
265 cgh.host_task([=]() {
266 auto write_bit_variant = old_page.write_bit();
267 auto& write_bit_future
268 = std::get<std::future<write_bit_t>>(write_bit_variant);
269 *shared_write_bit = write_bit_future.get();
270 });
271 });
272 auto write_bit_event
273 = queue.submit([&, write_bit_retrieval_event](sycl::handler& cgh) {
274 cgh.depends_on(write_bit_retrieval_event);
275 cgh.host_task([=]() mutable {
276 new_page.set_write_bit(*shared_write_bit);
277 });
278 });
279 *this = std::move(new_page);
280 return queue.submit([copy_event, write_bit_event](sycl::handler& cgh) {
281 cgh.depends_on(std::move(copy_event));
282 cgh.depends_on(std::move(write_bit_event));
283 });
284 }

294 static auto should_skip_copy(
295 const page_handle& dst,
296 const page_handle& src,
297 page_copy_rules rules
298 ) -> std::variant<bool, std::future<bool>> {
299 if (rules.expect_valid_dst) {
300 if (!dst.is_valid()) {
301 throw std::runtime_error(
302 "Destination page is invalid in copy_from"
303 );
304 }
305 } else {
306 if (dst.is_valid()) {
307 return true;
308 }
309 }
310
311 if (rules.expect_valid_src) {
312 if (!src.is_valid()) {
313 throw std::runtime_error("Source page is invalid in copy_from");
314 }
315 } else {
316 if (!src.is_valid()) {
317 return true;
318 }
319 }
320
321 if (rules.ignore_unwritten) {
322 auto write_bit_variant = src.write_bit();
323 if (src.is_mpi_local()) {
324 auto write_bit = std::get<write_bit_t>(write_bit_variant);
325 if (write_bit == 0) {
326 return true;
327 }
328 } else {
329 auto& write_bit_future
330 = std::get<std::future<write_bit_t>>(write_bit_variant);
331 return std::async(
332 std::launch::async,
333 [write_bit_future = std::move(write_bit_future)]() mutable {
334 return write_bit_future.get() == 0;
335 }
336 );
337 }
338 }
339
340 if (!rules.ignore_page_index) {
341 if (dst.index() != src.index()) {
342 throw std::runtime_error(
343 "Page indices do not match in copy_from"
344 );
345 }
346 }
347
348 return false;
349 }

370template<class T, page_size_t PageSize = default_page_size>
371struct page_traits {
372 static constexpr page_size_t page_size = PageSize;
373 static constexpr page_size_t elements_per_page = page_size / sizeof(T);
374 static constexpr page_size_t allocated_bytes_per_page
375 = elements_per_page * sizeof(T);
376 static_assert(elements_per_page > 0, "Page size is too small for type T");
377};
378
379template<class T, page_size_t PageSize = default_page_size>
380constexpr auto required_pages_for_elements(page_count_t elements)
381 -> page_count_t {
382 return (elements + page_traits<T, PageSize>::elements_per_page - 1)
383 / page_traits<T, PageSize>::elements_per_page;
384}

386template<
387 class T,
388 class PageBeginIterator,
389 class PageEndIterator,
390 class PageDstIterator,
391 page_size_t PageSize = default_page_size>
392auto copy_pages(
393 const sycl::queue& queue,
394 PageBeginIterator begin,
395 PageEndIterator end,
396 PageDstIterator dst,
397 page_copy_rules rules = {}
398) -> sclx::event {
399 std::vector<sclx::event> copy_events(std::distance(begin, end));
400 std::transform(
401 begin,
402 end,
403 copy_events.begin(),
404 [queue, dst, rules](auto& page) {
405 return page.copy_from(*dst++, rules);
406 }
407 );
408
409 return queue.submit([copy_events
410 = std::move(copy_events)](sycl::handler& cgh) {
411 cgh.depends_on(copy_events);
412 });
413}
