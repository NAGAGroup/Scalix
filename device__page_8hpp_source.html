82 auto copy_from(sycl::queue queue, std::shared_ptr<page_interface> src)
83 -> sclx::event override {
84 auto src_device_id_variant = src->device_id();
85 device_id_t src_device_id = 0;
86 if (src->is_mpi_local()) {
87 src_device_id = std::get<device_id_t>(src_device_id_variant);
88 if (src_device_id == no_device) {
89 throw std::runtime_error(
90 "Source page has no device associated with it, cannot "
91 "copy from it. This error indicates a bug in Scalix, "
92 "please "
93 "report it."
94 );
95 }
96 } else {
97 throw std::runtime_error(
98 "Device pages cannot copy from MPI devices yet."
99 );
100 }
101
102 if (device_id_ == no_device) {
103 throw std::runtime_error(
104 "Device page has no device associated with it, cannot copy "
105 "to it. This error indicates a bug in Scalix, please report "
106 "it."
107 );
108 }
109
110 auto src_data_variant = src->data();
111 auto src_data = std::get<sclx::byte*>(src_data_variant);
112 if (src_data == data_) {
113 return {};
114 }
115 return queue.submit([&](sycl::handler& cgh) {
116 cgh.memcpy(data_, src_data, allocated_bytes_per_page_);
117 });
118 }

124 write_bit_t write_bit_{0};
125 device_id_t device_id_{no_device};
126 page_index_t index_{0};
127 page_size_t allocated_bytes_per_page_{0};
128};
