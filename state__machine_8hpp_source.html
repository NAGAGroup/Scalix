172template<class StateDefinition>
173 requires StateDefinitionType<StateDefinition>
174class state_machine {
175 public:
176 using state_type = StateDefinition;
177 using inputs_type =
178 typename state_machine_traits<state_machine>::inputs_type;
179
180 void transition_states(const inputs_type& inputs) {
181 static constexpr auto invalid_label_value
182 = state_machine_traits<state_machine>::invalid_state;
183 static constexpr auto terminal_label_value
184 = state_machine_traits<state_machine>::terminal_state;
185 auto new_state = state_type::make_invalid_state();
186
187 if (current_state_->label() == terminal_label_value) {
188 current_state_ = std::move(new_state);
189 return;
190 }
191
192 for (const auto& transition : current_state_->possible_transitions()) {
193 if (auto transition_result
194 = transition->execute(inputs, *current_state_);
195 transition_result->label() != invalid_label_value) {
196 new_state = new_state->label() == invalid_label_value
197 ? std::move(transition_result)
198 : throw std::runtime_error(
199 "Multiple transitions possible, which is "
200 "not allowed"
201 );
202 }
203 }
204 current_state_ = std::move(new_state);
205 }
206
207 auto current_state() const -> const
208 typename state_machine_traits<state_machine>::label_type& {
209 return current_state_->label();
210 }
211
212 private:
213 std::unique_ptr<state_type> current_state_{state_type::make_initial_state()
214 };
215};
